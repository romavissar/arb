```typescript
// Extract cadence logic into a testable pure function
function computeCycleMode(
  totalCycles: number,
  discoveryIntervalCycles: number,
  shortTermPassCycles: number,
  hasMatchedPairs: boolean
): "discovery" | "short-term" | "refresh" {
  const doDiscovery = !hasMatchedPairs || totalCycles % discoveryIntervalCycles === 0;
  if (doDiscovery) return "discovery";
  const doShortTermPass = totalCycles % shortTermPassCycles === 0;
  if (doShortTermPass) return "short-term";
  return "refresh";
}
```
